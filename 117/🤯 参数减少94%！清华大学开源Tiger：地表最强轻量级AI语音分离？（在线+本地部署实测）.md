# 安装指南

## 安装编译工具
sudo apt update && sudo apt upgrade -y  
sudo apt install build-essential -y  

## 创建运行环境
conda create -n tiger python=3.10 -y  
conda activate tiger  

## 克隆项目
git clone https://github.com/JusperLee/TIGER.git  
cd TIGER  

## 安装依赖组件
pip install -r requirements.txt  
pip install gradio  

## 推理
python app.py
